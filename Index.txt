    function saveGiftPositions() {
        giftPositions = [];
        for (let r = 0; r < 8; r++) {
            for (let c = 0; c < 8; c++) {
                if (board[r][c] === '🎁') {
                    giftPositions.push({ r, c });
                }
            }
        }
    }

    function typeEffect(text) {
        clearInterval(typingTimer);
        storyText.innerText = "";
        let i = 0;
        typingTimer = setInterval(() => {
            if (i < text.length) { storyText.innerText += text.charAt(i); i++; }
            else { clearInterval(typingTimer); }
        }, 30);
    }

    function onMainButtonClick() {
        if (storyText.innerText.length < fullText.length) {
            clearInterval(typingTimer);
            storyText.innerText = fullText;
            return;
        }

        const btnText = mainBtn.innerText;
        if (btnText === "次のステージへ") {
            currentStage++;
            updateBackground();
            timeLeft = 60;
            isGameFinished = false;
            setupOverlay(false); // 次のステージの「開始画面」を表示
        } else if (btnText === "タイトルへ戻る" || btnText === "最初から遊ぶ") {
            currentStage = 0; timeLeft = 60; score = 0; scoreText.innerText = "0";
            giftPositions = [];
            updateBackground();
            showTitleScreen();
        } else if (btnText === "物語を始める" || btnText === "ゲームを始める") {
            startGame();
        }
    }

    function onRetryButtonClick() {
        timeLeft = 60; score = 0; scoreText.innerText = "0";
        giftPositions = [];
        startGame();

        document.getElementById('quote-text').innerText = quoteObj.t;
    }

    function startTimer() {
        clearInterval(gameInterval);

            timeLeft -= 0.1;

    }

        const p = (timeLeft / 60) * 100;
        timerBar.style.width = p + "%";
        timeText.innerText = Math.ceil(timeLeft);
    }

        const panels = STAGES[currentStage].panels;
        const index = panels.indexOf(emoji);
        if (emoji === '🎁') return 'var(--gift-pink)';
        return index !== -1 ? COLOR_PALETTE[index % COLOR_PALETTE.length] : '#fff';
    }

        list.innerHTML = '';
        missions = {};
        
        let targets;
        if (stage.isFinal) {
            targets = ['🥧', '📜', '👱🏻‍♀️'];
        } else {
            targets = [stage.panels[0], stage.panels[stage.panels.length-1]];
        }
        
        targets.forEach(emoji => {
            missions[emoji] = 12;
            const item = document.createElement('div');
            item.className = 'mission-item';
            item.id = 'mission-' + emoji;
            item.style.backgroundColor = getPanelColor(emoji);
            item.innerHTML = `<span>${emoji}</span>: <span class="m-count">${missions[emoji]}</span>`;
            list.appendChild(item);
        });
        document.getElementById('stage-name').innerText = stage.name;
        document.getElementById('stage-num-label').innerText = `Stage ${currentStage + 1}`;
    }

    function initBoard() {
        const panels = STAGES[currentStage].panels;
        boardEl.innerHTML = '';
        board = [];
        for (let r = 0; r < 8; r++) {
            board[r] = [];
            for (let c = 0; c < 8; c++) {
                const isGiftPos = giftPositions.some(pos => pos.r === r && pos.c === c);
                if (isGiftPos) {
                    board[r][c] = '🎁';
                    continue;
                }
                let p;
                do { p = panels[Math.floor(Math.random() * panels.length)]; }
                while ((r >= 2 && board[r-1][c] === p && board[r-2][c] === p) || (c >= 2 && board[r][c-1] === p && board[r][c-2] === p));
                board[r][c] = p;
            }
        }
        render(true);
    }

    function render(isFullInit = false) {
        if (isFullInit) boardEl.innerHTML = '';
        for (let r = 0; r < 8; r++) {
            for (let c = 0; c < 8; c++) {
                let t = document.getElementById(`tile-${r}-${c}`);
                if (isFullInit && !t) {
                    t = document.createElement('div');
                    t.id = `tile-${r}-${c}`;
                    t.dataset.r = r; t.dataset.c = c;
                    boardEl.appendChild(t);
                }
                const p = board[r][c];
                t.innerText = p || "";
                t.className = 'tile';
                t.style.backgroundColor = getPanelColor(p);
                if(p === '🎁') t.classList.add('gift-panel');
                else t.classList.remove('gift-panel');
            }
        }
        updateGlow();
    }

    function updateGlow() {
        const activeMissions = Object.keys(missions).filter(k => missions[k] > 0);
        const allTiles = document.querySelectorAll('.tile');
        allTiles.forEach(t => t.classList.remove('last-target-glow'));
        if (activeMissions.length === 1) {
            allTiles.forEach(t => { if (t.innerText === activeMissions[0]) t.classList.add('last-target-glow'); });
        }
    }

    function onStart(e) {
        if (isProcessing || isGameFinished) return;
        const t = e.target.closest('.tile');
        if (!t) return;
        document.querySelectorAll('.hint-highlight').forEach(el => el.classList.remove('hint-highlight'));
        if (t.innerText === '🎁') { useGift(parseInt(t.dataset.r), parseInt(t.dataset.c)); return; }
        dragStart = { r: parseInt(t.dataset.r), c: parseInt(t.dataset.c), x: e.clientX || e.pageX, y: e.clientY || e.pageY, el: t };
        t.classList.add('selected');
    }

    async function useGift(r, c) {
        isProcessing = true;
        const currentTypes = [...new Set(board.flat())].filter(p => p && p !== '🎁');
        if (currentTypes.length === 0) { isProcessing = false; return; }
        const targetType = currentTypes[Math.floor(Math.random() * currentTypes.length)];
        board[r][c] = null;
        document.getElementById(`tile-${r}-${c}`).classList.add('eliminating');
        for (let i = 0; i < 8; i++) {
            for (let j = 0; j < 8; j++) {
                if (board[i][j] === targetType) {
                    if (missions[targetType] !== undefined && missions[targetType] > 0) missions[targetType]--;
                    score += 15; board[i][j] = null;
                    document.getElementById(`tile-${i}-${j}`).classList.add('eliminating');
                }
            }
        }
        scoreText.innerText = score; updateMissionUI();
        await new Promise(res => setTimeout(res, 400));
        dropTiles(); render(); await processMatches();
    }

    function onMove(e) {
        if (!dragStart || isProcessing || isGameFinished) return;
        const curX = e.clientX || e.pageX; const curY = e.clientY || e.pageY;
        const dx = curX - dragStart.x; const dy = curY - dragStart.y;
        if (Math.abs(dx) > 25 || Math.abs(dy) > 25) {
            let tr = dragStart.r, tc = dragStart.c;
            if (Math.abs(dx) > Math.abs(dy)) tc += dx > 0 ? 1 : -1;
            else tr += dy > 0 ? 1 : -1;
            if (tr >= 0 && tr < 8 && tc >= 0 && tc < 8) {
                const r1 = dragStart.r, c1 = dragStart.c;
                dragStart.el.classList.remove('selected'); dragStart = null;
                handleSwap(r1, c1, tr, tc);
            }
        }
    }

    function onEnd() { if (dragStart) { dragStart.el.classList.remove('selected'); dragStart = null; } }

    async function handleSwap(r1, c1, r2, c2) {
        isProcessing = true;
        const p1 = board[r1][c1], p2 = board[r2][c2];
        board[r1][c1] = p2; board[r2][c2] = p1; render();
        const matches = findMatches();
        if (matches.length > 0) { await processMatches(); }
        else { await new Promise(res => setTimeout(res, 200)); board[r1][c1] = p1; board[r2][c2] = p2; render(); isProcessing = false; }
    }

    function findMatches() {
        let m = [];
        for (let r = 0; r < 8; r++) {
            for (let c = 0; c < 6; c++) {
                if (board[r][c] && board[r][c] !== '🎁' && board[r][c] === board[r][c+1] && board[r][c] === board[r][c+2]) m.push({r,c}, {r,c:c+1}, {r,c:c+2});
            }
        }
        for (let c = 0; c < 8; c++) {
            for (let r = 0; r < 6; r++) {
                if (board[r][c] && board[r][c] !== '🎁' && board[r][c] === board[r+1][c] && board[r][c] === board[r+2][c]) m.push({r,c}, {r:r+1,c}, {r:r+2,c});
            }
        }
        return m.filter((v, i, a) => a.findIndex(t => t.r === v.r && t.c === v.c) === i);
    }

    async function processMatches() {
        if (isGameFinished) return;
        let matches = findMatches();
        while (matches.length > 0) {
            matches.forEach(pos => {
                const p = board[pos.r][pos.c];
                if (p) {
                    if (missions[p] !== undefined && missions[p] > 0) missions[p]--;
                    score += 10; removedTotal++;
                    document.getElementById(`tile-${pos.r}-${pos.c}`).classList.add('eliminating');
                    board[pos.r][pos.c] = null;
                }
            });
            if (removedTotal >= GIFT_THRESHOLD) { removedTotal -= GIFT_THRESHOLD; spawnGift(); }
            scoreText.innerText = score; updateMissionUI();
            await new Promise(res => setTimeout(res, 350));
            dropTiles(); render();
            await new Promise(res => setTimeout(res, 200));
            matches = findMatches();
        }
        checkClearStatus(); isProcessing = false;
    }

    function checkClearStatus() {
        if (!isGameFinished && Object.values(missions).every(v => v <= 0)) {
            isGameFinished = true; clearInterval(gameInterval);
            setTimeout(() => setupOverlay(true), 500);
        }
    }

    function spawnGift() {
        let r, c; do { r = Math.floor(Math.random() * 8); c = Math.floor(Math.random() * 8); } while (board[r][c] === '🎁');
        board[r][c] = '🎁';
    }

    function dropTiles() {
        const panels = STAGES[currentStage].panels;
        for (let c = 0; c < 8; c++) {
            let empty = 0;
            for (let r = 7; r >= 0; r--) {
                if (board[r][c] === null) empty++;
                else if (empty > 0) { board[r+empty][c] = board[r][c]; board[r][c] = null; }
            }
            for (let r = 0; r < empty; r++) { board[r][c] = panels[Math.floor(Math.random() * panels.length)]; }
        }
    }

    function updateMissionUI() {
        for (const emoji in missions) {
            const el = document.getElementById('mission-' + emoji);
            if (el) el.querySelector('.m-count').innerText = Math.max(0, missions[emoji]);
        }
    }

    function updateHintBtn() {
        const b = document.getElementById('hint-btn');
        b.innerText = `💡 ヒント (${hintCount})`;
        b.disabled = hintCount <= 0;
    }

    function useHint() {
        if (hintCount <= 0 || isProcessing || isGameFinished) return;
        for (let r = 0; r < 8; r++) {
            for (let c = 0; c < 8; c++) {
                if (c < 7 && checkPotentialMatch(r, c, r, c + 1)) { highlight(r, c, r, c + 1); hintCount--; updateHintBtn(); return; }
                if (r < 7 && checkPotentialMatch(r, c, r + 1, c)) { highlight(r, c, r + 1, c); hintCount--; updateHintBtn(); return; }
            }
        }
    }

    function checkPotentialMatch(r1, c1, r2, c2) {
        const p1 = board[r1][c1]; const p2 = board[r2][c2];
        if (!p1 || !p2 || p1 === '🎁' || p2 === '🎁' || p1 === p2) return false;
        board[r1][c1] = p2; board[r2][c2] = p1;
        const m = findMatches();
        board[r1][c1] = p1; board[r2][c2] = p2;
        return m.length > 0;
    }

    function highlight(r1, c1, r2, c2) {
        const t1 = document.getElementById(`tile-${r1}-${c1}`);
        const t2 = document.getElementById(`tile-${r2}-${c2}`);
        t1.classList.add('hint-highlight'); t2.classList.add('hint-highlight');
        setTimeout(() => { t1.classList.remove('hint-highlight'); t2.classList.remove('hint-highlight'); }, 3000);
    }

    function createConfetti() {
        for (let i = 0; i < 30; i++) {
            const c = document.createElement('div');
            c.className = 'confetti';
            c.style.left = Math.random() * 100 + 'vw';
            c.style.backgroundColor = `hsl(${Math.random() * 360}, 100%, 50%)`;
            c.style.animationDelay = Math.random() * 2 + 's';
            storyOverlay.appendChild(c);
            setTimeout(() => c.remove(), 3000);
        }
    }

    function gameOver() { clearInterval(gameInterval); clearInterval(quoteInterval); setupOverlay(false); }
</script>
